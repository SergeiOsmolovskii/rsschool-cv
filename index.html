function sortByBit(arr) {
    let countBit = (bit) => bit.toString(2).replace(/[0]/g,'').length;
    arr.sort((a, b) => {
        if (countBit(a) === countBit(b)) return a - b;
        else return countBit(a) - countBit(b);
    });
}
