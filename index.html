function validParentheses(par) {
    const openBracket = ['(','[','{']; 
    const closeBracket = [')',']','}'];
    let stack = [];
    for(let i=0; i < par.length; i++) {
        if (openBracket.includes(par[i])) {
        stack.push(par[i]);
        } else {
            if (closeBracket.indexOf(par[i]) === openBracket.indexOf(stack[stack.length-1])) {
                stack.pop();
            } else return false; 
        }   
    }
    if (stack.length == 0) {
        return true;
    } else return false;
}
